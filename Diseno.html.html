430      // ✅ Cargar HTML divididos + ejecutar scripts internos (para Galeria)
431      async function loadInto(id, file) {
432          const container = document.getElementById(id);
433          try {
434              const res = await fetch(file, { cache: "no-store" });
435              if (!res.ok) throw new Error("HTTP " + res.status);
436  
437              container.innerHTML = await res.text();
438  
439              // Ejecutar scripts dentro del HTML cargado
440              const scripts = container.querySelectorAll("script");
441              scripts.forEach(oldScript => {
442                  const newScript = document.createElement("script");
443                  if (oldScript.src) newScript.src = oldScript.src;
444                  else newScript.textContent = oldScript.textContent;
445                  oldScript.parentNode.replaceChild(newScript, oldScript);
446              });
447  
448              // Reaplicar idioma a lo que entró
449              applyLanguage(currentLang);
450          } catch (err) {
451              console.error("Error cargando " + file + ":", err);
452              container.innerHTML =
453                  '<div class="card"><p style="text-align:center;color:var(--text-muted)">' +
454                  'No se pudo cargar <b>' + file + '</b>. (Debe estar en la misma carpeta y abrir con servidor / Open in Browser)' +
455                  '</p></div>';
456          }
457      }

461      loadInto("menuContent", "Menu.html");
462      loadInto("galeriaContent", "Galeria.html");

464      // Idioma inicial
465      applyLanguage("de");
