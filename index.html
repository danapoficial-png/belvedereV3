        function applyTranslation(lang) {
            if (!translations[lang]) return;
            const currentTranslations = translations[lang];

            // 1. Traducir elementos con IDs (Incluye ahora todos los elementos del menú)
            for (const selector in currentTranslations) {
                const element = document.querySelector(selector);
                if (element) {
                    if (selector === '#tel-info' || selector === '#email-info') {
                        // Para elementos que necesitan HTML (e.g. el enlace telefónico)
                        element.innerHTML = currentTranslations[selector];
                    }
                    else if (element.tagName === 'INPUT' || element.tagName === 'BUTTON') {
                        // Para botones e inputs
                        element.textContent = currentTranslations[selector];
                    }
                    else {
                        // Para todos los demás elementos (texto simple)
                        element.textContent = currentTranslations[selector];
                    }
                }
            }

            // 2. Ajustar el texto del botón "ver más"
            const isVisible = miniBio.classList.contains('visible');
            let showMoreText = currentTranslations['#ver-mas-text'];
            let showLessText = translations[lang]['#h-closed-mon'] === 'Geschlossen' ? 'Weniger anzeigen' :
                lang === 'it' ? 'Meno' :
                    lang === 'fr' ? 'Moins' : 'Show Less';

            verMasText.textContent = isVisible ? showLessText : showMoreText;
        }
